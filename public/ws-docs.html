WebSocket API

Bidirectional real-time connection for voice room events. All messages are JSON.

Connection Endpoint

io("http://localhost:3000", { auth: { userId: "userId", roomId: "roomId" } })
userId — Your user ID (from POST /users)
roomId — The room ID (from POST /rooms or GET /rooms)
npm install socket.io-client

The server validates that both the room and user exist before upgrading the connection. Returns HTTP 400/404 if invalid.

Typical Flow

Client ──────────────────────────────────────── Server
1. POST /users → get userId
2. GET /rooms or POST /rooms → get roomId
3. WS connect →
← { type: "connected" }
← { type: "room.snapshot", data: {...} }
4. { type: "join_room" } →
← { type: "ack", command: "join_room", status: "ok" }
← { type: "event", event: "user_joined", ... } broadcast to room
5. { type: "take_seat", seatIndex: 1 } →
← { type: "ack", command: "take_seat", status: "ok" }
← { type: "event", event: "mic_changed", ... } broadcast to room
6. { type: "leave_room" } →
← { type: "ack", command: "leave_room", status: "ok" }

Client → Server Commands

Send JSON messages over the WebSocket connection. Every command gets an ack or error response.

C→S ping Keep-alive heartbeat ▶ 
{
  "type": "ping"
}

// Response:
{ "type": "pong", "roomId": "...", "userId": "..." }
C→S join_room Join the room as audience. If approval required, returns status: "pending". ▶ 
{
  "type": "join_room"
}

// Response (direct join):
{ "type": "ack", "command": "join_room", "status": "ok",
  "data": { "id": "...", "roomId": "...", "userId": "...", "memberRole": "audience", "seatIndex": -1 } }

// Response (approval required):
{ "type": "ack", "command": "join_room", "status": "pending",
  "data": { "id": "...", "roomId": "...", "userId": "...", "status": "pending" } }

// Broadcast to room on success:
{ "type": "event", "event": "user_joined",
  "data": { "userId": "...", "nickname": "Alice", "memberRole": "audience" } }
C→S leave_room Leave the room and free any held seat ▶ 
{
  "type": "leave_room"
}

// Response:
{ "type": "ack", "command": "leave_room", "status": "ok" }

// Broadcast to room:
{ "type": "event", "event": "user_left",
  "data": { "userId": "...", "nickname": "Alice" } }
C→S take_seat Occupy a specific seat (go on mic). seatIndex 0 = host seat. ▶ 
{
  "type": "take_seat",
  "seatIndex": 1   // integer, 0 = host seat, 1..N = guest seats
}

// Response:
{ "type": "ack", "command": "take_seat", "status": "ok",
  "data": { "id": "...", "memberRole": "speaker", "seatIndex": 1, ... } }

// Broadcast to room:
{ "type": "event", "event": "mic_changed",
  "data": { "userId": "...", "nickname": "Alice", "seatIndex": 1, "memberRole": "speaker" } }

// Errors:
{ "type": "error", "command": "take_seat", "error": "Seat 1 is already occupied" }
{ "type": "error", "command": "take_seat", "error": "Seat 1 is locked" }
{ "type": "error", "command": "take_seat", "error": "User is not in this room" }
C→S leave_seat Leave current seat (go off mic) ▶ 
{
  "type": "leave_seat"
}

// Response:
{ "type": "ack", "command": "leave_seat", "status": "ok" }

// Broadcast to room:
{ "type": "event", "event": "mic_changed",
  "data": { "userId": "...", "nickname": "Alice", "seatIndex": -1, "memberRole": "audience" } }
C→S toggle_mic Shortcut: auto-assign first idle seat (speaker) or leave seat (audience) ▶ 
{
  "type": "toggle_mic",
  "role": "speaker"   // "speaker" | "audience"
}

// Same response/broadcast as take_seat / leave_seat
C→S send_comment Post a comment or reply (max 500 chars). Must be in the room. ▶ 
// Top-level comment:
{ "type": "send_comment", "content": "Hello everyone!" }

// Reply to a comment:
{ "type": "send_comment", "content": "Agreed!", "parentId": "comment-uuid" }

// Response:
{ "type": "ack", "command": "send_comment", "status": "ok",
  "data": { "id": "...", "roomId": "...", "userId": "...", "content": "Hello everyone!",
            "parentId": null, "replyCount": 0, "nickname": "Alice", "createdAt": 1713600000000 } }

// Broadcast to room:
{ "type": "event", "event": "message",
  "data": { "id": "...", "userId": "...", "nickname": "Alice", "content": "Hello everyone!",
            "parentId": null, "parentNickname": null, "createdAt": 1713600000000 } }
C→S kick_member Kick a member from the room. Owner only. ▶ 
{
  "type": "kick_member",
  "targetUserId": "user-uuid"
}

// Response:
{ "type": "ack", "command": "kick_member", "status": "ok" }

// Sent to kicked user:
{ "type": "event", "event": "member_kicked",
  "data": { "userId": "...", "nickname": "Bob" } }

// Broadcast to room:
{ "type": "event", "event": "member_kicked",
  "data": { "userId": "...", "nickname": "Bob" } }
C→S close_room Close the room permanently. Owner only. ▶ 
{
  "type": "close_room"
}

// Response:
{ "type": "ack", "command": "close_room", "status": "ok" }

// Broadcast to ALL room members:
{ "type": "event", "event": "room_closed",
  "data": { "roomId": "..." } }

Server → Client Messages

S→C connected Sent immediately after WebSocket handshake ▶ 
{
  "type": "connected",
  "roomId": "room-uuid",
  "userId": "user-uuid"
}
S→C room.snapshot Full room state sent right after connected. Use this to initialize UI. ▶ 
{
  "type": "room.snapshot",
  "roomId": "room-uuid",
  "data": {
    "id": "room-uuid",
    "name": "Chill Vibes",
    "ownerId": "user-uuid",
    "maxSpeakers": 8,
    "status": "active",
    "requireApproval": true,
    "announcement": null,
    "members": [
      { "id": "...", "userId": "...", "nickname": "Alice",
        "memberRole": "audience", "seatIndex": -1, "joinedAt": 1713600000000 }
    ],
    "seats": [
      { "roomId": "...", "seatIndex": 0, "userId": null, "nickname": null,
        "status": "idle", "micEnabled": true },
      { "roomId": "...", "seatIndex": 1, "userId": null, "nickname": null,
        "status": "idle", "micEnabled": true }
    ],
    "onlineCount": 1
  }
}
S→C ack Success response to a command ▶ 
{
  "type": "ack",
  "command": "join_room",   // the command that was executed
  "status": "ok",           // "ok" | "pending"
  "data": { ... }           // command-specific result object (optional)
}
S→C error Error response to a command, or connection-level error ▶ 
{
  "type": "error",
  "command": "take_seat",   // which command failed (omitted for connection errors)
  "error": "Seat 1 is already occupied"
}

Real-time Events

All events share the same envelope. Some events are targeted (only sent to a specific user), others are broadcast to the entire room.

{
  "type": "event",
  "event": "<event_type>",
  "roomId": "room-uuid",
  "data": { ... }   // event-specific payload
}
eventdataRecipientsDescription
user_joined { userId, nickname, memberRole } All Someone joined the room
user_left { userId, nickname } All Someone left the room
mic_changed { userId, nickname, seatIndex, memberRole } All User went on/off mic. seatIndex=-1 means off mic.
seat_locked { seatIndex, locked } All Owner locked or unlocked a seat
seat_muted { seatIndex, muted } All Owner muted or unmuted a seat
room_closed { roomId } All Room was closed by owner. Disconnect and redirect.
member_kicked { userId, nickname } All + kicked user A member was kicked. Kicked user receives it first.
join_request { requestId, userId, nickname, avatarUrl } Owner only Someone requested to join. Approve or reject via REST API.
join_approved { requestId, roomId, roomName } Requester only Join request was approved. Send join_room command now.
join_rejected { requestId, roomId, roomName } Requester only Join request was rejected.
message { id, userId, nickname, content, parentId, parentNickname, createdAt } All New comment or reply posted in the room

Data Schemas

RoomMember

{
  "id":         string,   // member record ID
  "roomId":     string,
  "userId":     string,
  "memberRole": "speaker" | "audience",
  "seatIndex":  number,   // -1 = not on a seat
  "joinedAt":   number,   // unix ms
  "nickname":   string,   // included in detail responses
  "avatarUrl":  string
}

SeatState

{
  "roomId":     string,
  "seatIndex":  number,           // 0 = host seat
  "userId":     string | null,    // null = idle
  "nickname":   string | null,
  "status":     "idle" | "occupied" | "locked" | "muted",
  "micEnabled": boolean
}

Comment / Reply

{
  "id":             string,
  "roomId":         string,
  "userId":         string,
  "parentId":       string | null,   // null = top-level comment
  "content":        string,
  "createdAt":      number,          // unix ms
  "nickname":       string,
  "avatarUrl":      string,
  "replyCount":     number,          // top-level only
  "parentNickname": string | null    // replies only
}

JoinRequest

{
  "id":        string,
  "roomId":    string,
  "userId":    string,
  "status":    "pending" | "approved" | "rejected",
  "createdAt": number
}

Client Example

// npm install socket.io-client
import { io } from "socket.io-client";

const socket = io("http://localhost:3000", {
  auth: { userId: "USER_ID", roomId: "ROOM_ID" }
});

// ── Server → Client ──────────────────────────────────────────────────────────

socket.on("connect", () => console.log("Connected:", socket.id));
socket.on("connect_error", (err) => console.error("Auth failed:", err.message));

socket.on("connected",     ({ roomId, userId }) => console.log("Session ready"));
socket.on("room.snapshot", ({ data }) => initUI(data));   // members + seats

socket.on("ack",   ({ command, status, data }) => console.log(command, status, data));
socket.on("error", ({ command, error }) => showError(error));
socket.on("pong",  () => console.log("pong"));

// Real-time events
socket.on("event", ({ event, data }) => {
  switch (event) {
    case "user_joined":   addMember(data);           break;
    case "user_left":     removeMember(data.userId); break;
    case "mic_changed":   updateSeat(data);          break;
    case "room_closed":   redirectToLobby();         break;
    case "member_kicked": if (data.userId === myId) redirectToLobby(); break;
    case "join_approved": socket.emit("join_room");  break;
    case "message":       appendComment(data);       break;
  }
});

// ── Client → Server ──────────────────────────────────────────────────────────

socket.emit("join_room");
socket.emit("take_seat",    { seatIndex: 1 });
socket.emit("leave_seat");
socket.emit("toggle_mic",   { role: "speaker" });
socket.emit("send_comment", { content: "Hello!", parentId: null });
socket.emit("kick_member",  { targetUserId: "user-uuid" });
socket.emit("close_room");
socket.emit("ping");
ws.send(JSON.stringify({ type: 'send_comment', content: 'Hello!' }));
